package main
 
import "fmt"
 
type person struct {
name string
age int
}
func newPerson(name string) *person {
p := person{ name: name }
p.age = 42
return &p
}
func (p *person) fmt1() string {
return p.name + " formatted"
}
func (p person) fmt2() string {
return p.name + " formatted 2"
}
func main() {
fmt.Println("Start experiments...")
fmt.Println(*newPerson("Alex"))
fmt.Println(person{ "Test", 39 })
fmt.Println(person{ name: "Test2", age: 35 })
p1 := person{ age: 31 }
p2 := &p1
p2.name = "xxx"
fmt.Println(p1, p2)
fmt.Println(p2.fmt1())
fmt.Println(p2.fmt2())
}
